  339.   function handleRequests() {

  342.       $this->stack=array($this->arg=$this->simpleArg);

  358.   }

  397.   function begin($name, $st) {
  398.     $arg = $this->arg;
  399.     switch($name[0]) {
  400.     case 'A':                        /* receive apply args as normal array */
  401.       $object = $this->globalRef->get($this->getExact($st['v']));
  402.       $newArg = new java_ApplyArg($this, 'A',
  403.                                   $this->parser->getData($st['m']),
  404.                                   $this->parser->getData($st['p']),
  405.                                   $object,
  406.                                   $this->getExact($st['n']));
  407.       $this->link($arg, $newArg);
  408.       array_push($this->stack, $this->arg = $newArg);
  409.       break;
  410.     case 'X': 
  411.       $newArg = new java_CompositeArg($this, $st['t']);
  412.       $this->link($arg, $newArg);
  413.       array_push($this->stack, $this->arg = $newArg);
  414.       break;
  415.     case 'P':
  416.       if($arg->type=='H') { /* hash table */
  417.         $s = $st['t'];
  418.         if(<a href="../JavaBridge/_Options.inc.html#defineJAVA_DEBUG">JAVA_DEBUG</a>) {echo "setresult prepare hash:"; echo sprintf("%s", $st['t']); echo "\n";}
  419.         if($s[0]=='N') { /* number */
  420.           $arg->setIndex($this->getExact($st['v']));
  421.           if(<a href="../JavaBridge/_Options.inc.html#defineJAVA_DEBUG">JAVA_DEBUG</a>) {echo "setresult array: index:"; echo sprintf("%s", $st['v']); echo "\n";}
  422.         } else {
  423.           $arg->setIndex($this->parser->getData($st['v']));
  424.           if(<a href="../JavaBridge/_Options.inc.html#defineJAVA_DEBUG">JAVA_DEBUG</a>) {echo "setresult hash: index:"; echo sprintf("%s", $this->parser->getData($st['v'])); echo "\n";}
  425.         }
  426.       } else {                    /* array */
  427.         $arg->setNextIndex();
  428.       }
  429.       break;
  430.     case 'S':
  431.       $arg->setResult($this->parser->getData($st['v']));
  432.       if(<a href="../JavaBridge/_Options.inc.html#defineJAVA_DEBUG">JAVA_DEBUG</a>) {echo "setresult string:"; echo sprintf("%s", $this->parser->getData($st['v'])); echo "\n";}
  433.       break;
  434.     case 'B':
  435.       $s=$st['v'];
  436.       $arg->setResult($s[0]=='T');
  437.       if(<a href="../JavaBridge/_Options.inc.html#defineJAVA_DEBUG">JAVA_DEBUG</a>) {echo "setresult bool:"; echo sprintf("%s", $st['v']); echo "\n";}
  438.       break;
  439.     case 'L':                    // unsigned long
  440.       $sign = $st['p'];
  441.       $val = $this->getExact($st['v']);
  442.       if($sign[0]=='A') $val*=-1;
  443.       $arg->setResult($val);
  444.       if(<a href="../JavaBridge/_Options.inc.html#defineJAVA_DEBUG">JAVA_DEBUG</a>) {echo "setresult long:"; echo sprintf("%s, sign: %s", $st['v'], $st['p']); echo "\n";}
  445.       break;
  446.     case 'D':
  447.       $arg->setResult($this->getInexact($st['v']));
  448.       if(<a href="../JavaBridge/_Options.inc.html#defineJAVA_DEBUG">JAVA_DEBUG</a>) {echo "setresult double:"; echo sprintf("%s", $st['v']); echo "\n";}
