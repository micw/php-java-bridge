  68. class java_SimpleParser {
  69.   public $SLEN=256; // initial length of the parser string
  70.  
  71.   public $handler;
  72.   
  73.   public $tag, $buf, $len, $s;
  74.   public $type;
  75.  
  76.   function java_SimpleParser($handler) {
  77.     $this->handler = $handler;
  78.     $this->tag = array(new java_ParserTag(), new java_ParserTag(), new java_ParserTag());
  79.     $this->len = $this->SLEN;
  80.     $this->s = str_repeat(" ", $this->SLEN);
  81.     $this->type = $this->VOJD;
  82.   }
  83.   
  84.   // VOJD is VOID for f... windows (VOID is in winsock2.h)
  85.   public $BEGIN=0, $KEY=1, $VAL=2, $ENTITY=3, $VOJD=5, $END=6; 
  86.   public $level=0, $eor=0; public $in_dquote, $eot=false;
  87.   public $pos=0, $c=0, $i=0, $i0=0, $e;
  88.  
  89.   function RESET() {
  90.     $this->type=$this->VOJD;
  91.     $this->level=0;
  92.     $this->eor=0;
  93.     $this->in_dquote=false;
  94.     $this->i=0;
  95.     $this->i0=0;
  96.   }
  97.     
  98.   function APPEND($c) {
  99.     if($this->i>=$this->len-1) {
  100.       $this->s=str_repeat($this->s,2);
  101.       $this->len*=2;
  102.     } 
  103.     $this->s[$this->i++]=$c; 
  104.   }
  105.   function CALL_BEGIN() {
  106.     $pt=&$this->tag[1]->strings;
  107.     $st=&$this->tag[2]->strings;
  108.     $t=&$this->tag[0]->strings[0];
  109.     $name=$t->string[$t->off];
  110.     $n = $this->tag[2]->n;
  111.     $ar = array();
  112.     for($i=0; $i<$n; $i++) {
  113.       $ar[$pt[$i]->getString()] = $st[$i]->getString();
  114.     }
  115.     $this->handler->begin($name, $ar);
  116.   }
  117.   function CALL_END() {
  118.     $t=&$this->tag[0]->strings[0];
  119.     $name=$t->string[$t->off];
  120.     $this->handler->end($name);
  121.   }
  122.   function PUSH($t) { 
  123.     $str = &$this->tag[$t]->strings;
  124.     $n = &$this->tag[$t]->n;
  125.     $this->s[$this->i]='|';
  126.     if(!isset($str[$n])){$h=$this->handler; $str[$n]=$h->createParserString();}
  127.     $str[$n]->string=&$this->s;
  128.     $str[$n]->off=$this->i0;
  129.     $str[$n]->length=$this->i-$this->i0;
  130.     ++$this->tag[$t]->n;
  131.     $this->APPEND('|');
  132.     $this->i0=$this->i;
  133.   }
  134.   function parse() {
  135.     while($this->eor==0) {
  136.       if($this->c>=$this->pos) { 
  137.         
  138.         $this->buf=$this->handler->read(JAVA_RECV_SIZE); 
  139.         if(is_null($this->buf) || strlen($this->buf) == 0) die("protocol error. Check the back end log for OutOfMemoryErrors...");
  140.         $this->pos=strlen($this->buf);
  141.         if($this->pos==0) break;
  142.         $this->c=0; 
  143.       }
  144.       switch(($ch=$this->buf[$this->c])) 
  145.         {/* --- This block must be compilable with an ansi C compiler or javac --- */
  146.         case '<': if($this->in_dquote) {$this->APPEND($ch); break;}
  147.           $this->level+=1;
  148.           $this->type=$this->BEGIN;
  149.           break;
  150.         case '\t': case '\f': case '\n': case '\r': case ' ': if($this->in_dquote) {$this->APPEND($ch); break;}
  151.           if($this->type==$this->BEGIN) {
  152.             $this->PUSH($this->type); 
  153.             $this->type = $this->KEY; 
  154.           }
  155.           break;
  156.         case '=': if($this->in_dquote) {$this->APPEND($ch); break;}
  157.           $this->PUSH($this->type);
  158.           $this->type=$this->VAL;
  159.           break;
  160.         case '/': if($this->in_dquote) {$this->APPEND($ch); break;}
  161.           if($this->type==$this->BEGIN) { $this->type=$this->END; $this->level-=1; }
  162.           $this->level-=1;
  163.           $this->eot=true; // used for debugging only
  164.           break;
  165.         case '>': if($this->in_dquote) {$this->APPEND($ch); break;}
  166.           if($this->type==$this->END){
  167.             $this->PUSH($this->BEGIN);
  168.             $this->CALL_END();
  169.           } else {
  170.             if($this->type==$this->VAL) $this->PUSH($this->type);
  171.             $this->CALL_BEGIN();
  172.           }
  173.           $this->tag[0]->n=$this->tag[1]->n=$this->tag[2]->n=0; $this->i0=$this->i=0;              /* RESET */
  174.           $this->type=$this->VOJD;
  175.           if($this->level==0) $this->eor=1; 
  176.           break;
  177.         case ';':
  178.           if($this->type==$this->ENTITY) {
  179.             switch ($this->s[$this->e+1]) {
  180.             case 'l': $this->s[$this->e]='<'; $this->i=$this->e+1; break; /* lt */
  181.             case 'g': $this->s[$this->e]='>'; $this->i=$this->e+1; break; /* gt */
  182.             case 'a': $this->s[$this->e]=($this->s[$this->e+2]=='m'?'&':'\''); $this->i=$this->e+1; break; /* amp, apos */
  183.             case 'q': $this->s[$this->e]='"'; $this->i=$this->e+1; break; /* quot */
  184.             default: $this->APPEND($ch);
  185.             }
  186.             $this->type=$this->VAL; //& escapes may only appear in values
  187.           } else {
  188.             $this->APPEND($ch);
  189.           }
  190.           break;
  191.         case '&': 
  192.           $this->type = $this->ENTITY;
  193.           $this->e=$this->i;
  194.           $this->APPEND($ch);
  195.           break;
  196.         case '"':
  197.           $this->in_dquote = !$this->in_dquote;
  198.           if(!$this->in_dquote && $this->type==$this->VAL) {
  199.             $this->PUSH($this->type);
  200.             $this->type = $this->KEY;
  201.           }
  202.           break;
  203.         default:
  204.           $this->APPEND($ch);
  205.         } /* ------------------ End of ansi C block ---------------- */
  206.       $this->c+=1;
  207.     }
  208.        $this->RESET();
  209.   }
