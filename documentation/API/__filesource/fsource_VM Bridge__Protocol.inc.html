  253.   function write($data) {
  254.

  256.       ? chr(0103-JAVA_PREFER_VALUES)

  271.     $res .= $webapp;
  272.     $res .= " HTTP/1.0\r\n";
  273.     $res .= "Host: localhost\r\n";
  274.     $res .= "Content-Length: "; $res .= $len; $res .= "\r\n";

  283.     $count = fwrite($socket, $res); fflush($socket);
  284.     return $count;
  285.   }

  292.   function parseHeaders() {
  293.     $this->headers = array();
  294.     while (($str = trim(fgets($this->socket, JAVA_RECV_SIZE)))) {
  295.       if($str[0]=='X') {
  296.         if(!strncasecmp("X_JAVABRIDGE_REDIRECT", $str, 21)) {
  297.           $this->headers["redirect"]=trim(substr($str, 22));
  298.         } else if(!strncasecmp("X_JAVABRIDGE_CONTEXT", $str, 20)) {
  299.           $this->headers["context"]=trim(substr($str, 21));
  300.         }
  301.       } else if($str[0]=='S') {    // Set-Cookie:
  302.         if(!strncasecmp("SET-COOKIE", $str, 10)) {
  303.           $str=substr($str, 12);
  304.           $ar = explode(";", $str);
  305.           $cookie = explode("=",$ar[0]);
  306.           $path = "";
  307.           if(isset($ar[1])) $p=explode("=", $ar[1]);
  308.           if(isset($p)) $path=$p[1];
  309.           $this->doSetCookie($cookie[0], $cookie[1], $path);
  310.         }
  311.       } else if($str[0]=='C') { // Content-Length
  312.         if(!strncasecmp("CONTENT-LENGTH", $str, 14)) {
  313.           $this->headers["content_length"]=trim(substr($str, 15));
  314.         }
  315.       }
  316.     }
  317.   }

  335.   function redirect() {
  336.     if(!isset($this->headers["redirect"])) { // no redirect received: must continue to send PUT requests
  337.       throw new java_RuntimeException("no Pipe- or SocketContextServer available.");

  339.     if(!isset($this->protocol->socketHandler)) {
  340.       $hostVec = java_Protocol::getHost();
  341.       $this->host = $hostVec[0];
  342.       $channelName = $this->headers["redirect"];
  343.       $context = $this->headers["context"];
  344.       $len = strlen($context);
  345.       $len0 = chr(0xFF);
  346.       $len1 = chr($len&0xFF); $len>>=8;
  347.       $len2 = chr($len&0xFF);
  348.       $this->protocol->socketHandler=new java_SocketHandler($this->protocol, $this->getChannel($channelName));
  349.       $this->protocol->write("\177${len0}${len1}${len2}${context}");
  350.       $this->context = sprintf("X_JAVABRIDGE_CONTEXT: %s\r\n", $context);
  351.     }
  352.     $this->close();
  353.     $this->protocol->handler = $this->protocol->socketHandler;
  354.     $this->protocol->flush(); // this write-write-read is necessary; otherwise the ContextFactory may time out
  355.   }

  383.   function createHttpHandler() {
  384.     $hostVec = java_Protocol::getHost();
  385.     $host = $hostVec[0];
  386.     $port = $hostVec[1];
  387.  
  388.     $overrideHosts = $this->getOverrideHosts();
  389.     $ssl = "";
  390.     if($overrideHosts) {
  391.       // handle "s:127.0.0.1:8080//JavaBridge/test.phpjavabridge" 
  392.       // or "s:127.0.0.1:8080" 
  393.       // or "/" 
  394.       // or ""
  395.      $s=$overrideHosts;
  396.      if((strlen($s)>2) && ($s[1]==':')) {
  397.        if($s[0]=='s')
  398.          $ssl="ssl://";
  399.        $s = substr($s, 2);
  400.      }
  401.      $webCtx = strpos($s, "//");
  402.      if($webCtx)
  403.        $host = substr($s, 0, $webCtx);
  404.      else
  405.        $host = $s;
  406.  
  407.      $idx = strpos($host, ':');
  408.      if($idx) {
  409.        if($webCtx)
  410.          $port = substr($host, $idx+1, $webCtx);
  411.        else
  412.          $port = substr($host, $idx+1);
  413.        $host = substr($host, 0, $idx);
  414.      } else {
  415.        $port = "8080";
  416.      }
  417.
