  46. class java_NativeParser {
  47.   public $parser, $handler;
  48.   public $level, $event;
  49.  
  50.   public $buf;
  51.  
  52.   function java_NativeParser($handler) {
  53.     $this->handler = $handler;
  54.     $this->parser = xml_parser_create();
  55.     xml_parser_set_option($this->parser, XML_OPTION_CASE_FOLDING, 0);
  56.     xml_set_object($this->parser, $this);
  57.     xml_set_element_handler($this->parser, "begin", "end");
  58.     xml_parse($this->parser, "<F>");
  59.     $this->level = 0;
  60.   }
  61.   function begin($parser, $name, $param) {
  62.       if (false) $parser = $parser;
  63.       
  64.     $this->event = true;
  65.     switch($name) {
  66.     case 'X': case 'A': $this->level+=1;
  67.     }
  68.     $this->handler->begin($name, $param);
  69.   }
  70.   function end($parser, $name) {
  71.       if (false) $parser = $parser;
  72.       
  73.     $this->handler->end($name);
  74.     switch($name) {
  75.     case 'X': case 'A': $this->level-=1;
  76.     }
  77.   }
  78.   function getData($str) {
  79.     return base64_decode($str);
  80.   }
  81.   function parse() {
  82.     // NOTE: RECV_SIZE must be the same as the underlying 
  83.     // PHP stream buffer size, otherwise this function will hang.
  84.     do {
  85.       $this->event = false;
  86.       $buf = $this->buf = $this->handler->read(JAVA_RECV_SIZE); 
  87.       $len = strlen($buf);
  88.       if(!xml_parse($this->parser, $buf, $len==0)) {
  89.         die(sprintf("protocol error: %s, %s at col %d. Check the back end log for details.",
  90.                     $buf,
  91.                     xml_error_string(xml_get_error_code($this->parser)),
  92.                     xml_get_current_column_number($this->parser)));
  93.       }
  94.     } while(!$this->event || $this->level>0);
  95.   }
  96.  
  97.   function parserError() {
  98.     die(sprintf("protocol error: %s. Check the back end log for details.", $this->buf));
  99.   }
  100. }
